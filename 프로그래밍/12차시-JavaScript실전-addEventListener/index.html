STEP 1 · classList

◆ classList — CSS 클래스를 JS로 제어

style.속성 대신 CSS 클래스를 켜고 끄는 더 좋은 방법

.add("클래스")

클래스를 추가합니다

요소에 새 옷 입히기

.remove("클래스")

클래스를 제거합니다

입던 옷 벗기기

.toggle("클래스")

있으면 제거, 없으면 추가

스위치처럼 켜고 끄기

BAD — style.속성 방식
// 여러 줄에 걸쳐 하나씩 변경 el.style.backgroundColor = "#1A1A2E"; el.style.color = "#fff"; el.style.padding = "20px";
GOOD — classList 방식
// CSS에 미리 정의하고, JS는 한 줄! el.classList.toggle("dark");
CSS에 스타일을 미리 정의하고, JS는 클래스 이름만 켜고 끄면 됩니다

STEP 1 · classList

✦ classList.toggle 실습

버튼을 클릭하면 하이라이트가 토글되는 예제

CSS 클래스 정의
.highlight { background: #E8FF3B; font-weight: bold; padding: 8px 16px; border-radius: 8px; }
JS 토글 코드
let text = document.getElementById("myText"); let btn = document.getElementById("hlBtn"); btn.addEventListener("click", function() { text.classList.toggle("highlight"); });

STEP 2 · 다크 모드

◆ 다크 모드 토글 만들기

HTML
<body> <h1>나의 웹페이지</h1> <button id="darkBtn">다크 모드</button> </body>
CSS
/* 기본: 밝은 배경 */ body { background: #F8F8F2; color: #1A1A2E; transition: 0.3s; } /* 다크 모드: 어두운 배경 */ .dark-mode { background: #1A1A2E; color: #ffffff; }
JS
let darkBtn = document.getElementById("darkBtn"); darkBtn.addEventListener("click", function() { document.body.classList.toggle("dark-mode"); });

1

3

CSS 자동 반영

CSS가 클래스에 맞춰 배경색·글자색 전환

JS는 클래스만 토글, 실제 스타일은 CSS가 담당!

CHALLENGE
다크 모드 버튼에 아이콘 전환 기능 추가
다크 모드일 때는 버튼에 "☀️ 라이트 모드"를, 라이트 모드일 때는 "🌙 다크 모드"를 표시하세요.
힌트: classList.contains("dark-mode")로 현재 상태 확인
darkBtn.addEventListener("click", function() { document.body.classList.toggle("dark-mode"); // 👇 여기를 채워보세요! if (document.body.classList.contains("dark-mode")) { darkBtn.textContent = "______"; // ☀️ 라이트 모드 } else { darkBtn.textContent = "______"; // 🌙 다크 모드 } });
classList.contains()는 해당 클래스가 있으면 true, 없으면 false를 반환합니다

여러 요소에 이벤트 등록
STEP 4 · FAQ 아코디언

◆ 여러 요소에 이벤트 한꺼번에 등록하기

"FAQ 질문이 5개인데, addEventListener를 5번 써야 하나요?"

BAD — 하나씩 등록
// 5개면 5번 반복... q1.addEventListener("click", ...); q2.addEventListener("click", ...); q3.addEventListener("click", ...); // ... 계속 반복 😩
GOOD — 한꺼번에 등록
// 몇 개든 한 번에! let questions = document .querySelectorAll(".faq-question"); questions.forEach(function(el) { el.addEventListener("click", function() { // 클릭 시 동작 }); });
querySelectorAll은 같은 클래스의 모든 요소를 배열처럼 가져오고, forEach로 각각에 이벤트를 등록합니다

STEP 4 · FAQ 아코디언

✦ 아코디언 CSS — 숨기기와 보이기

HTML 구조
<div class="faq-item"> <div class="faq-question"> Q. 수업 준비물이 뭔가요? </div> <div class="faq-answer"> 노트북과 VS Code만 있으면 됩니다! </div> </div>
CSS 핵심 트릭
/* 기본: 답변 숨김 */ .faq-answer { max-height: 0; overflow: hidden; transition: max-height 0.3s; } /* active 클래스가 붙으면: 답변 표시 */ .faq-item.active .faq-answer { max-height: 200px; }
동작 원리

숨김 상태

max-height: 0 + overflow: hidden으로 높이를 0으로 만들어 숨김

표시 상태

.active 클래스가 붙으면 max-height: 200px으로 펼침

부드러운 애니메이션

transition이 높이 변화를 부드럽게 처리

let questions = document .querySelectorAll(".faq-question"); questions.forEach(function(q) { q.addEventListener("click", function() { // 부모 .faq-item 선택 let item = q.parentElement; // active 클래스 토글 item.classList.toggle("active"); }); });
